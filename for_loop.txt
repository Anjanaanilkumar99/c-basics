1)program to print sum of n numbers

#include <stdio.h>
int main() 
{
    int i,sum,n;
    printf("enter a number ");
    scanf("%d",&n);
    for(i=1;i<=n;i++)
      { 
         sum=sum+i;
       }
     printf("sum is=%d\n",sum);

    return 0;
}
